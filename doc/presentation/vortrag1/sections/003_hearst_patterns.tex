\begin{frame}
  \frametitle{Hearst Patterns: Regelen}

  $$R \subseteq M \times M$$
  $$\forall x, y, z \in M : xRy \land yRz \Rightarrow xRz$$
  $$\forall x \in M : \neg xRx$$
  $$\forall x, y \in M : xRy \Rightarrow \neg (yRx)$$
\end{frame}
